Responsive JSF Layout Template /** * Toggles the visibility of the navigation sidebar. If it is visible, it will be hidden and if it * was hidden, it will be shown. When showing or hiding the sidebar, the main content area will be * modified as well in order to reflect the latest changes in screen width. */ function toggle_sidebar_nav() { var sidebarNav = document.getElementById("layoutSidebar"); var mainContent = document.getElementById("mainContent"); if (sidebarNav.style.display != "none") { sidebarNav.style.display = "none"; mainContent.style.marginLeft = "0px"; } else { sidebarNav.style.display = "block"; sidebarNav.style.width = "#{profile.sidebarWidth}px"; mainContent.style.marginLeft = "#{profile.sidebarWidth}px"; } }
